# Windows Activity Monitor - Configuration File
# 使用此配置文件需要将 src/main_advanced.rs 重命名为 src/main.rs

# 在 https://discord.com/developers/applications 获取

# 更新间隔（秒）

# 是否显示时间戳
SHOW_TIMESTAMPS=true

# 是否显示应用图标
SHOW_ICONS=true
